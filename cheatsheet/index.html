Node Cheat Sheet

Control Flow

Callback pattern

function foo(val1, val2, callback) {
  ...
  callback();
}

Promise pattern

function foo(ok) {
  return new Promise(resolve, reject) {
    if (ok) {
      resolve('success');
    } else {
      reject('boo');
    }
  }
}

foo()
  .then(res => {
    console.log(res);
  })
  .catch(err => {
    throw err;
  })

Async/Await

async function callFoo() {
  try {
    const result = await foo(true);
  } catch(err) {
    throw err;
  }
}
